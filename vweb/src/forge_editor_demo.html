        const str = "Hello, world!";

        fn main() {
          repeat(1200) n {
            poke(0x10000 + 1200 + n, 0x6f);
            poke(0x10000 + n, 0);
          }

          var n = 0;
          while(peek(str + n)) {
            poke(0x10000 + 41 + n, peek(str + n));
            n = n + 1;
          }
        }
